// Updated Firebase Firestore Rules
// Copy and paste these rules into your Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Bookings collection
    match /bookings/{bookingId} {
      // Anyone can create a booking (public form)
      allow create: if true;
      
      // Only authenticated users can read, update, delete (admin dashboard)
      allow read, update, delete: if request.auth != null;
    }
    
    // Users collection - NEW
    match /users/{userId} {
      // Only authenticated users can read users list
      allow read: if request.auth != null;
      
      // Only authenticated users can create, update, or delete users
      // In production, you might want to restrict this to admin role only:
      // allow write: if request.auth != null && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
      allow create, update, delete: if request.auth != null;
    }
  }
}
